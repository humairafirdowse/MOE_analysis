\section{Parameter Counting (Overview Tab)}

\subsection{Per-Component Parameter Counts}

\paragraph{Embedding}
\[
P_{\text{embed}} = V \cdot d_{\text{model}}.
\]

\paragraph{Per-Layer Attention with GQA}
Let $H$ be the number of attention heads, $H_{\text{kv}}$ the number of KV heads,
and $d_{\text{head}}$ the per-head dim. Projections:
\begin{align*}
P_Q &= d_{\text{model}} \cdot (H d_{\text{head}}), \\
P_K &= d_{\text{model}} \cdot (H_{\text{kv}} d_{\text{head}}), \\
P_V &= d_{\text{model}} \cdot (H_{\text{kv}} d_{\text{head}}), \\
P_O &= (H d_{\text{head}}) \cdot d_{\text{model}}.
\end{align*}
Per layer:
\[
P_{\text{attn,layer}} = P_Q + P_K + P_V + P_O.
\]
All layers:
\[
P_{\text{attn}} = L \cdot P_{\text{attn,layer}}.
\]

\paragraph{Per-Layer Attention with MLA (DeepSeek-V2/V3)}
Multi-head Latent Attention uses low-rank projections. Let $d_c$ (kv\_lora\_rank),
$d'_c$ (q\_lora\_rank), $d^R_h$ (qk\_rope\_head\_dim), $d^{\text{nope}}_h$ (qk\_nope\_head\_dim),
$d_v$ (v\_head\_dim). Per layer:
\begin{align*}
P_{Q_a} &= d_{\text{model}} \cdot d'_c, \quad
P_{Q_b} = d'_c \cdot (H \cdot (d^{\text{nope}}_h + d^{\text{rope}}_h)), \\
P_{\text{KV},a} &= d_{\text{model}} \cdot (d_c + d^R_h), \quad
P_{\text{KV},b} = d_c \cdot (H \cdot (d^{\text{nope}}_h + d_v)), \\
P_O &= (H \cdot d_v) \cdot d_{\text{model}}.
\end{align*}
\[
P_{\text{attn,layer}} = P_{Q_a} + P_{Q_b} + P_{\text{KV},a} + P_{\text{KV},b} + P_O.
\]

\paragraph{Per-Expert FFN (SwiGLU)}
For both routed and shared experts, assuming SwiGLU:
\[
P_{\text{ffn,expert}} = 3 \cdot d_{\text{model}} \cdot d_{\text{ff}},
\quad
P_{\text{ffn,shared}} = 3 \cdot d_{\text{model}} \cdot d_{\text{ff,sh}}.
\]

\paragraph{Routed and Shared Experts}
With $L_{\text{moe}}$ MoE layers (UI: $L_{\text{moe}} = L - K_{\text{dense}}$ when $K_{\text{dense}} > 0$, else $L_{\text{moe}} = L$):
\begin{align*}
P_{\text{experts}} &= L_{\text{moe}} \cdot E \cdot P_{\text{ffn,expert}}, \\
P_{\text{shared}}  &= L_{\text{moe}} \cdot N_{\text{sh}} \cdot P_{\text{ffn,shared}}.
\end{align*}

\paragraph{Dense FFN Layers (e.g.\ DeepSeek-V3)}
First $K_{\text{dense}}$ layers use dense FFN (SwiGLU) with intermediate dim $d_{\text{ff,dense}}$:
\[
P_{\text{dense}} = K_{\text{dense}} \cdot 3 \, d_{\text{model}} \, d_{\text{ff,dense}}.
\]
These layers are always active; $L_{\text{moe}}$ excludes them.

\paragraph{Residual MLP}
Each MoE layer may add a parallel dense MLP with intermediate $d_{\text{ff,res}}$:
\[
P_{\text{res,MLP}} = L_{\text{moe}} \cdot 3 d_{\text{model}} d_{\text{ff,res}}.
\]

\paragraph{Gating / Router Parameters}
Each MoE layer uses a linear router $d_{\text{model}} \to E$ (only MoE layers have gating):
\[
P_{\text{gate,layer}} = d_{\text{model}} \cdot E,
\quad
P_{\text{gate}} = L_{\text{moe}} \cdot P_{\text{gate,layer}}.
\]

\paragraph{RMSNorm / LayerNorm Parameters}
Assuming two norms per layer (pre-attention, pre-FFN):
\[
P_{\text{norm,layer}} = 2 \cdot d_{\text{model}},
\quad
P_{\text{norm}} = L \cdot P_{\text{norm,layer}}.
\]

\paragraph{Output Head}
We count the output projection even if tied with embedding:
\[
P_{\text{out}} = V \cdot d_{\text{model}}.
\]

\subsection{Total and Active Parameters}

\paragraph{Total Parameters}
\[
P_{\text{total}} =
P_{\text{embed}} + P_{\text{attn}} + P_{\text{experts}} + P_{\text{shared}}
+ P_{\text{gate}} + P_{\text{norm}} + P_{\text{out}}
+ P_{\text{dense}} + P_{\text{res,MLP}},
\]
where $P_{\text{dense}}$ and $P_{\text{res,MLP}}$ are zero when not used.

\paragraph{Active Parameters per Token}
Only $K$ experts fire per MoE layer per token; shared experts,
gating, norms, dense layers and residual MLP are always active:
\begin{align*}
P_{\text{experts,active}} &= L_{\text{moe}} \cdot K \cdot P_{\text{ffn,expert}}, \\
P_{\text{shared,active}}  &= P_{\text{shared}}, \\
P_{\text{gate,active}}    &= P_{\text{gate}}, \\
P_{\text{norm,active}}    &= P_{\text{norm}}, \\
P_{\text{dense,active}}   &= P_{\text{dense}}, \quad
P_{\text{res,MLP,active}} = P_{\text{res,MLP}}.
\end{align*}
Then
\[
P_{\text{active}} =
P_{\text{embed}} + P_{\text{attn}} +
P_{\text{experts,active}} + P_{\text{shared,active}}
+ P_{\text{gate,active}} + P_{\text{norm,active}} + P_{\text{out}}
+ P_{\text{dense,active}} + P_{\text{res,MLP,active}}.
\]

The equivalent dense baseline used in the UI has
\[
P_{\text{dense,eq}} = P_{\text{active}}.
\]

\section{Training Compute (FLOPs Tab)}

\subsection{Attention FLOPs per Token with GQA}

For sequence length $S$:

Projection FLOPs per layer, per token:
\begin{align*}
F_Q &= 2 d_{\text{model}} (H d_{\text{head}}), \\
F_K &= 2 d_{\text{model}} (H_{\text{kv}} d_{\text{head}}), \\
F_V &= 2 d_{\text{model}} (H_{\text{kv}} d_{\text{head}}), \\
F_O &= 2 (H d_{\text{head}}) d_{\text{model}},
\end{align*}
\[
F_{\text{proj}} = F_Q + F_K + F_V + F_O.
\]

Attention scores and context:
\[
F_{\text{score}}   = 2 S H d_{\text{head}}, \quad
F_{\text{context}} = 2 S H d_{\text{head}},
\]
\[
F_{\text{attn,compute}} = F_{\text{score}} + F_{\text{context}}
= 4 S H d_{\text{head}}.
\]

Per layer, per token:
\[
F_{\text{attn,layer}} = F_{\text{proj}} + F_{\text{attn,compute}}.
\]
All layers:
\[
F_{\text{attn}} = L \cdot F_{\text{attn,layer}}.
\]

When $H_{\text{kv}} = H$ and $H d_{\text{head}} = d_{\text{model}}$,
this reduces to the classical
\[
F_{\text{attn,layer}} \approx
8 d_{\text{model}}^2 + 4 S d_{\text{model}}.
\]

\subsection{Attention FLOPs with MLA}
For MLA (DeepSeek-V2/V3), let $d_{\text{q,head}} = d^{\text{nope}}_h + d^{\text{rope}}_h$.
Projection FLOPs per layer:
\begin{align*}
F_{Q_a} &= 2 d_{\text{model}} d'_c, \quad
F_{Q_b} = 2 d'_c (H d_{\text{q,head}}), \\
F_{\text{KV},a} &= 2 d_{\text{model}} (d_c + d^R_h), \quad
F_{\text{KV},b} = 2 d_c (H (d^{\text{nope}}_h + d_v)), \\
F_O &= 2 (H d_v) d_{\text{model}}.
\end{align*}
\[
F_{\text{proj}} = F_{Q_a} + F_{Q_b} + F_{\text{KV},a} + F_{\text{KV},b} + F_O.
\]
Attention scores and context:
\[
F_{\text{score}} = 2 S H d_{\text{q,head}}, \quad
F_{\text{context}} = 2 S H d_v.
\]
\[
F_{\text{attn,layer}} = F_{\text{proj}} + F_{\text{score}} + F_{\text{context}}.
\]

\subsection{MoE FFN and Gating FLOPs per Token}

Per active expert (SwiGLU):
\[
F_{\text{ffn,expert,layer}} = 6 d_{\text{model}} d_{\text{ff}}.
\]

Per token across MoE layers:
\begin{align*}
F_{\text{ffn,moe}} &= L_{\text{moe}} \cdot K \cdot 6 d_{\text{model}} d_{\text{ff}}, \\
F_{\text{ffn,shared}} &= L_{\text{moe}} \cdot N_{\text{sh}} \cdot 6 d_{\text{model}} d_{\text{ff,sh}}.
\end{align*}

Gating per layer:
\[
F_{\text{gate,layer}} = 2 d_{\text{model}} E,
\quad
F_{\text{gate}} = L_{\text{moe}} \cdot F_{\text{gate,layer}}.
\]

\subsection{Total Forward and Dense-Equivalent FLOPs}

Total forward FLOPs per token (UI includes dense FFN and residual MLP when used):
\[
F_{\text{fwd,MoE}} = F_{\text{attn}} + F_{\text{ffn,moe}} + F_{\text{ffn,shared}} + F_{\text{gate}}
+ F_{\text{ffn,dense}} + F_{\text{ffn,res,MLP}},
\]
where $F_{\text{ffn,dense}} = K_{\text{dense}} \cdot 6 \, d_{\text{model}} \, d_{\text{ff,dense}}$ and
$F_{\text{ffn,res,MLP}} = L_{\text{moe}} \cdot 6 \, d_{\text{model}} \, d_{\text{ff,res}}$ when residual MLP is enabled.

Dense-equivalent forward (all experts fire in MoE layers; dense layers and residual MLP unchanged):
\[
F_{\text{ffn,dense,eq}} = L_{\text{moe}} \cdot E \cdot 6 \, d_{\text{model}} \, d_{\text{ff}},
\quad
F_{\text{fwd,dense}} =
F_{\text{attn}} + F_{\text{ffn,dense,eq}} + F_{\text{ffn,shared}} + F_{\text{ffn,dense}} + F_{\text{ffn,res,MLP}}.
\]

MoE vs dense compute ratio:
\[
r_{\text{MoE/dense}} = \frac{F_{\text{fwd,MoE}}}{F_{\text{fwd,dense}}}.
\]

\subsection{Training FLOPs and GPU-Hours}

Per-token training FLOPs: backward is $2 \times$ forward (UI: \texttt{backwardFlopsPerToken = 2 * forwardFlopsPerToken}), so
\[
F_{\text{train,token}} = F_{\text{fwd,MoE}} + 2 F_{\text{fwd,MoE}} = 3 \cdot F_{\text{fwd,MoE}}.
\]

Total for $T$ tokens:
\[
F_{\text{train,total}} = F_{\text{train,token}} \cdot T.
\]

Total GPU count:
\[
N_{\text{gpus}} = \mathrm{TP} \cdot \mathrm{EP} \cdot \mathrm{PP} \cdot \mathrm{DP}.
\]

Peak TFLOPS per GPU depends on training precision (UI): FP32 uses GPU FP32 peak; BF16 uses FP16 peak; FP8 uses \texttt{fp8TrainingTflops} or 2$\times$FP16. Effective per-GPU TFLOPS:
\[
F_{\text{eff,per-gpu}} = F_{\text{gpu}}(\text{precision}) \cdot U.
\]

Wall-clock time and GPU-hours:
\[
t_{\text{train}} = \frac{F_{\text{train,total}}}{F_{\text{eff,per-gpu}} \cdot N_{\text{gpus}} \cdot 10^{12}}
\quad [\text{s}],
\quad
\text{GPU-hours} = \frac{t_{\text{train}} \cdot N_{\text{gpus}}}{3600}
= \frac{F_{\text{train,total}}}{F_{\text{eff,per-gpu}} \cdot 10^{12} \cdot 3600}.
\]
Training cost (UI): $\text{cost} = \text{GPU-hours} \times \text{costPerGpuHour}$.

\section{Training Memory}

\subsection{Parameters, Optimizer, and Gradients}

Let $P_{\text{total}}$ be total parameters. For training precision with bytes $b_{\text{train}}$:
\[
M_{\text{params}} = P_{\text{total}} \cdot b_{\text{train}}.
\]

Optimizer states (UI: \texttt{optimizerBytesPerParam}):
\[
M_{\text{opt}} = P_{\text{total}} \cdot (\text{bytes per param}),
\quad
\text{bytes/param} =
\begin{cases}
8,  & \text{Adam, FP32 (m + v only)},\\
12, & \text{Adam, mixed precision (FP32 master + m + v)},\\
4,  & \text{AdaFactor-like}.
\end{cases}
\]

Gradient memory (UI: \texttt{gradPrecision}):
\[
M_{\text{grads}} =
\begin{cases}
4 \, P_{\text{total}}, & \text{FP32 gradients},\\
2 \, P_{\text{total}}, & \text{BF16 gradients}.
\end{cases}
\]

\subsection{Activation Memory}

Batch size in sequences (UI: \texttt{batchSizeSeq = globalBatchTokens / max(seqLen, 1)}):
\[
B_{\text{seq}} = \frac{B_{\text{global}}}{\max(S, 1)}.
\]

Storage at $b_{\text{train}}$ bytes/val:

Attention activations (pre-softmax logits + post-softmax weights). If FlashAttention is enabled, UI sets $M_{\text{attn,act}} = 0$; otherwise:
\[
M_{\text{attn,act}} =
2 \, B_{\text{seq}} \, H \, S^2 \, b_{\text{train}}.
\]

FFN activations (top-$K$ experts, at least one):
\[
M_{\text{ffn,act}} =
B_{\text{seq}} \, S \, d_{\text{ff}} \, \max(K, 1) \, b_{\text{train}}.
\]

Residual stream (two copies per layer in UI):
\[
M_{\text{resid}} =
2 \, B_{\text{seq}} \, S \, d_{\text{model}} \, b_{\text{train}}.
\]

Per layer:
\[
M_{\text{act,layer}} = M_{\text{attn,act}} + M_{\text{ffn,act}} + M_{\text{resid}}.
\]

\subsection{Checkpointing Modes}

UI uses $L$ layers and \texttt{activationCheckpointing}:
\begin{itemize}
  \item \textbf{None:} store all layers.
  \[
  M_{\text{act,total}} = L \cdot M_{\text{act,layer}}.
  \]

  \item \textbf{sqrt:} uniform $\sqrt{L}$ checkpoints.
  \[
  M_{\text{act,total}} = 2 \cdot \lceil \sqrt{L} \rceil \cdot M_{\text{act,layer}}.
  \]

  \item \textbf{Selective:} attention recomputed, FFN + residual stored per layer.
  \[
  M_{\text{act,total}} = L \cdot (M_{\text{ffn,act}} + M_{\text{resid}}).
  \]
\end{itemize}

\subsection{Peak and Per-GPU Memory}

Peak training memory:
\[
M_{\text{peak}} = M_{\text{params}} + M_{\text{opt}} + M_{\text{grads}} + M_{\text{act,total}}.
\]

Sharding (UI: weights/optimizer/gradients over TP$\times$EP$\times$PP; activations over TP$\times$PP only):
\begin{align*}
M_{\text{state,per-gpu}} &=
\frac{M_{\text{params}} + M_{\text{opt}} + M_{\text{grads}}}
{\max(\mathrm{TP} \cdot \mathrm{EP} \cdot \mathrm{PP}, 1)},\\
M_{\text{act,per-gpu}} &=
\frac{M_{\text{act,total}}}{\max(\mathrm{TP} \cdot \mathrm{PP}, 1)}, \\
M_{\text{per-gpu}} &= M_{\text{state,per-gpu}} + M_{\text{act,per-gpu}}.
\end{align*}

\section{Inference Analysis}

\subsection{FLOPs and Latency}

Prefill: per-token forward FLOPs at sequence length $S_{\text{in}}$ are $F_{\text{fwd,MoE}}(S_{\text{in}})$. Total prefill FLOPs for batch $B$:
\[
F_{\text{prefill,total}} = F_{\text{fwd,MoE}}(S_{\text{in}}) \cdot B \cdot S_{\text{in}}.
\]
UI uses effective prefill TFLOPS = $0.4 \times$ peak GPU TFLOPS (FP8/INT8 use FP8 peak when set). Compute GPUs for one request = TP $\times$ EP (PP stages sequential). Prefill latency:
\[
t_{\text{prefill}} = \frac{F_{\text{prefill,total}}}{(\mathrm{TP} \cdot \mathrm{EP}) \cdot F_{\text{eff,inf}} \cdot 10^{12}}
\quad [\text{s}],
\quad
t_{\text{prefill,ms}} = 1000 \, t_{\text{prefill}}.
\]
Prefill tokens/s = $B \cdot S_{\text{in}} / t_{\text{prefill}}$.

Decode: memory-bound. Per decode step the UI reads (sharded) active weights + KV cache. Non-expert active params sharded by TP; expert active params sharded by TP$\times$EP. Bytes per GPU per step:
\[
B_{\text{decode,per-gpu}} = \frac{P_{\text{non-expert,active}} \cdot b_{\text{inf}}}{\mathrm{TP}}
+ \frac{P_{\text{expert,active}} \cdot b_{\text{inf}}}{\mathrm{TP} \cdot \mathrm{EP}}
+ \frac{M_{\text{kv,total}}}{\mathrm{TP}}.
\]
Decode latency per token (per GPU): $t_{\text{decode}} = B_{\text{decode,per-gpu}} / \mathrm{BW}_{\text{mem}}$. Each step produces $B$ tokens (batch size), so
\[
\text{decode tokens/s} = \frac{B}{t_{\text{decode}}}.
\]
TTFT = $t_{\text{prefill}}$; inter-token latency = $t_{\text{decode}}$; total generation time = TTFT + $S_{\text{out}} \cdot t_{\text{decode}}$.

\subsection{Weights and KV Cache}

Weights:
\[
M_{\text{weights}} = P_{\text{total}} \cdot b_{\text{inf}}.
\]

KV cache per token (UI implementation):
\begin{itemize}
  \item \textbf{Standard (GQA/MHA):} key and value each $H_{\text{kv}} d_{\text{head}}$ per layer:
  \[
  M_{\text{kv,token}} = 2 \, L \, H_{\text{kv}} \, d_{\text{head}} \, b_{\text{inf}}.
  \]
  \item \textbf{MLA (DeepSeek-V2/V3):} single compressed latent per layer, $(d_c + d^R_h)$ elements:
  \[
  M_{\text{kv,token}} = L \, (d_c + d^R_h) \, b_{\text{inf}}.
  \]
  (UI uses one factor of $L (d_c + d^R_h)$, not $2L$.)
\end{itemize}

For batch size $B$ and effective sequence length $S_{\text{tot}}$:
\[
M_{\text{kv,total}} = M_{\text{kv,token}} \cdot B \cdot S_{\text{tot}}.
\]

Total inference memory:
\[
M_{\text{inf,total}} =
M_{\text{weights}} + M_{\text{kv,total}}.
\]

Per GPU, available bytes for KV after weights: $M_{\text{avail}} = \text{HBM}_{\text{GB}} \cdot 10^9 - M_{\text{weights,per-gpu}}$. KV-limited max batch size (UI):
\[
B_{\max} = \left\lfloor \frac{M_{\text{avail}}}{M_{\text{kv,token}} \cdot S_{\text{tot}} / \mathrm{TP}} \right\rfloor.
\]

\subsection{KV vs Sequence-Length Curve}

UI sequence-length sweep: for sample lengths $S_i$, KV cache (GB) = $M_{\text{kv,token}} \cdot B \cdot S_i / 10^9$; decode latency (ms) = $1000 \cdot B_{\text{decode,per-gpu}}(S_i) / \mathrm{BW}_{\text{mem}}$ with $B_{\text{decode,per-gpu}}$ including sharded active weights and KV for that $S_i$; TTFT at $S_i$ from prefill FLOPs at $S_i$; total time = TTFT + $S_{\text{out}} \cdot t_{\text{decode}}$.

\section{Parallelism and Communication}

\subsection{Total GPUs and Layout}

\[
N_{\text{gpus}} = \mathrm{TP} \cdot \mathrm{EP} \cdot \mathrm{PP} \cdot \mathrm{DP}.
\]

Experts and layers per GPU (UI):
\[
E_{\text{per-gpu}} = \frac{E}{\max(\mathrm{EP}, 1)},
\quad
L_{\text{per-gpu}} = \frac{L}{\max(\mathrm{PP}, 1)}.
\]

\subsection{Communication Volumes per Step}

$B_{\text{micro}}$ = sequences per DP replica per step = $B_{\text{global}} / (S \cdot \mathrm{DP})$ (UI: \texttt{microBatchSeq}).

\paragraph{All-to-All (EP)}
Only MoE layers incur expert-parallel traffic. Per MoE layer:
\[
C_{\text{all2all,layer}} =
2 \, B_{\text{micro}} \, S \, d_{\text{model}} \, b_{\text{train}}
\,\frac{\mathrm{EP}-1}{\mathrm{EP}}.
\]
Total (UI: \texttt{allToAllBytesPerStep}):
\[
C_{\text{all2all}} = L_{\text{moe}} \cdot C_{\text{all2all,layer}}.
\]

\paragraph{AllReduce (TP)}
All $L$ layers. Per layer (UI uses factor 4 for reduce-scatter + allgather):
\[
C_{\text{allred,TP,layer}} =
2 \cdot 2 \, B_{\text{micro}} \, S \, d_{\text{model}} \, b_{\text{train}}
\,\frac{\mathrm{TP}-1}{\mathrm{TP}}
= 4 \, B_{\text{micro}} \, S \, d_{\text{model}} \, b_{\text{train}}
\,\frac{\mathrm{TP}-1}{\mathrm{TP}}.
\]
\[
C_{\text{allred,TP}} = L \cdot C_{\text{allred,TP,layer}}.
\]

\paragraph{P2P (PP)}
\[
C_{\text{p2p,PP}} =
B_{\text{micro}} \, S \, d_{\text{model}} \, b_{\text{train}}
\cdot \max(\mathrm{PP}-1, 0).
\]

\paragraph{AllReduce (DP)}
Gradient sync once per step:
\[
C_{\text{allred,DP}} =
2 \, P_{\text{total}} \, b_{\text{train}}
\,\frac{\mathrm{DP}-1}{\mathrm{DP}}.
\]

\paragraph{Total}
\[
C_{\text{total}} =
C_{\text{all2all}} + C_{\text{allred,TP}} +
C_{\text{p2p,PP}} + C_{\text{allred,DP}}.
\]

\section{Routing and Efficiency}

\subsection{Active/Total Parameter Ratio}

Simple ratio used as an MoE utilisation proxy:
\[
r_{\text{active}} = \frac{K}{E}.
\]

\subsection{Gating Overhead}

UI uses full forward total (including dense FFN and residual MLP when present):
\[
\text{Overhead}_{\text{gate}} =
\frac{F_{\text{gate}}}{F_{\text{fwd,MoE}}}.
\]

\subsection{Expert Capacity and Drop Rate}

Tokens per expert (expected), for global batch tokens $B_{\text{global}}$:
\[
\lambda = \frac{B_{\text{global}} \cdot K}{E}.
\]
Capacity factor $C_{\text{cap}}$ (e.g.\ $1.25$) gives
\[
C = C_{\text{cap}} \cdot \lambda.
\]

Assuming a Poisson approximation for tokens per expert,
with $X \sim \mathrm{Poisson}(\lambda)$, the drop rate is
the tail probability
\[
p_{\text{drop}} \approx \Pr[X > C].
\]

For large $\lambda$ we use a normal approximation:
\[
p_{\text{drop}} \approx
1 - \Phi\!\left(\frac{C - \lambda}{\sqrt{\lambda}}\right)
= 1 - \Phi\!\left((C_{\text{cap}} - 1)\sqrt{\lambda}\right),
\]
where $\Phi$ is the standard normal CDF.

The UI implementation uses a rational approximation for the
normal tail to compute $p_{\text{drop}}$.

\subsection{MoE vs Dense FLOPs}

MoE FLOPs per token (UI: \texttt{flopsMoePerToken}):
\[
F_{\text{MoE,token}} = F_{\text{fwd,MoE}}.
\]

Dense (same total params): all experts fire in MoE layers (\texttt{flopsDenseTotalPerToken}):
\[
F_{\text{dense,total,token}} = F_{\text{fwd,dense}}.
\]

Dense (same active params): UI scales dense total FLOPs by active/total parameter ratio:
\[
F_{\text{dense,active,token}} = F_{\text{dense,total,token}} \cdot \frac{P_{\text{active}}}{P_{\text{total}}}.
\]
(\texttt{flopsDenseActivePerToken}.)

\section{Hardware Efficiency (UI)}

\subsection{Training: MFU, HFU, Bubble, Throughput}

Model FLOPs Utilization (user-set): $U = \mathrm{MFU}$.

HFU includes recomputation from activation checkpointing. Recompute multiplier (UI):
\begin{itemize}
  \item \textbf{None:} $r = 1$.
  \item \textbf{sqrt:} $r = 4/3$.
  \item \textbf{Selective:} $r = 1 + F_{\text{attn}} / (3 F_{\text{fwd,MoE}})$.
\end{itemize}
\[
\mathrm{HFU} = \mathrm{MFU} \cdot r.
\]

Pipeline bubble (1F1B): $\mathrm{numMicrobatches} = \lfloor B_{\text{global}} / (\mathrm{DP} \cdot S) \rfloor$,
\[
\text{bubble} = \frac{\mathrm{PP} - 1}{\mathrm{numMicrobatches}} \times 100\%.
\]

Effective throughput (tokens/s/GPU): $F_{\text{eff,per-gpu}} = F_{\text{gpu}} \cdot U$,
\[
\text{tokens/s/GPU} = \frac{F_{\text{eff,per-gpu}} \cdot 10^{12}}{3 \cdot F_{\text{fwd,MoE}}}.
\]

Compute-comm overlap: compute time per step = $\text{computeFlopsPerStep} / (F_{\text{eff,per-gpu}} \cdot 10^{12})$; comm time = $C_{\text{total}} / \mathrm{BW}_{\text{interconnect}}$; overlap \% = $\min(\text{compute}, \text{comm}) / \max(\text{compute}, \text{comm}) \times 100$.

\subsection{Inference: Roofline}

Ridge point (FLOPs/byte): $\text{ridge} = (F_{\text{peak,TFLOPS}} \cdot 10^{12}) / \mathrm{BW}_{\text{mem}}$.

Arithmetic intensity: $\mathrm{AI} = \text{FLOPs} / \text{bytes accessed}$. Prefill: $\text{FLOPs} = F_{\text{fwd}}(S_{\text{in}}) \cdot B \cdot S_{\text{in}}$, bytes = full weights. Decode: FLOPs = $F_{\text{fwd}}(S_{\text{tot}})$ per token, bytes = active weights + KV cache for batch.

Achievable TFLOPS: $\min(F_{\text{peak}} \cdot 10^{12},\, \mathrm{BW}_{\text{mem}} \cdot \mathrm{AI}) / 10^{12}$. Bound = compute if $\mathrm{AI} \ge \text{ridge}$, else memory.

Decode GPU utilization (UI): $\text{decodeLatencySec} = M_{\text{active,weights}} / \mathrm{BW}_{\text{mem}}$; $\text{decodeActualFlops} = F_{\text{fwd}} / \text{decodeLatencySec}$; $\text{util\%} = \text{decodeActualFlops} / (F_{\text{peak}} \cdot 10^{12}) \times 100$.

\section{Cost Analysis (UI)}

\subsection{Training Cost}

$\text{costPerGpuHour}$ = user or preset GPU \$/hr.
\[
\text{Total training cost} = \text{GPU-hours} \times \text{costPerGpuHour}.
\]
\[
\text{Cost per trillion tokens} = \frac{\text{Total training cost}}{T / 10^{12}}.
\]
\[
\text{Wall-clock hours} = \frac{\text{GPU-hours}}{N_{\text{gpus}}},
\quad
\text{Wall-clock days} = \frac{\text{Wall-clock hours}}{24}.
\]

\subsection{Inference Cost}

Per 1M input tokens: $\text{costPer1MInput} = (10^6 / \text{prefillTokensPerSec}) \times (\text{costPerGpuHour} / 3600)$. Per 1M output tokens: $\text{costPer1MOutput} = (10^6 / \text{decodeTokensPerSec}) \times (\text{costPerGpuHour} / 3600)$.

Batch-size sweep: same formulas at each batch size; decode bytes and thus decode tokens/s depend on $B$. Quantization sweep: same at each precision (weights and bytes per param change; decode tokens/s and \$/1M output change).

\subsection{Monthly Serving Estimate}

GPUs needed: $\text{reqPerSec} = \text{reqPerDay} / 86400$, $\text{secPerReq} = \text{secPerInputToken} \cdot \text{avgInputTok} + \text{secPerOutputToken} \cdot \text{avgOutputTok}$, $\text{gpusNeeded} = \lceil \text{reqPerSec} \cdot \text{secPerReq} \rceil$. Daily cost = (inputTokensPerDay/1e6)$\cdot$costPer1MInput + (outputTokensPerDay/1e6)$\cdot$costPer1MOutput; monthly = daily $\times$ 30.

\section{Validation Metrics}

\subsection{Deviation and Status}

For each metric with paper value $x_{\text{paper}}$ and
tool estimate $x_{\text{ours}}$, the relative deviation is
\[
\delta = \frac{x_{\text{ours}} - x_{\text{paper}}}{x_{\text{paper}}}.
\]

Status thresholds:
\[
\text{PASS} \iff |\delta| \le 0.05,
\quad
\text{WARN} \iff 0.05 < |\delta| \le 0.15,
\quad
\text{FAIL} \iff |\delta| > 0.15.
\]
